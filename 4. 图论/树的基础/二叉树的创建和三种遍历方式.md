typedef struct node{
    char data;
    struct node *lchild;
    struct node *rchild;
}BiTree;

BiTree *createTree(){
    char ch;
    BiTree *treenode;
    scanf("%c",&ch);
    if(ch=='#')//当前结点为空
        treenode=NULL;
    else{
        treenode=(BiTree*)malloc(sizeof(BiTree));
        treenode->data=ch;
        treenode->lchild=createTree();
        treenode->rchild=createTree();
    }
    return treenode;
}

using namespace std;

    }
}

void InOrderTraverse(BiTree *T){/*二叉树的中序遍历递归算法*/
